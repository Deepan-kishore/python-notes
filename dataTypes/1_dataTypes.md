# Data Types :

1. Numbers
- int: 1, -1
- float: 0.1, - 0.1
- complex: 2 + 6j

2. Sequences:
- string: 'charectersSequence'
- list : [1,'23','anyTypeAllowed']
- tuple : (2,'anyTypeAllowed')
- range: range(rangeParameter ) 
   - rangeParameter (
    startNumber: Optional ,
    endNumber: Mandatory,
    step: Optional //increment number between the range
    )

3. Mapping:
- dictionary: {a:2} //like Object in js

4. Set:
- set : {1,2,3}
- frozenSet : frozenset([1,2,3])

5. Boolean

6. Binary:
- bytes : b"ConvertedToByte"
- byteArray : byteArray([12,13,21]) // should be within 256
- memoryView : memoryview(b"ConvertedToByte")

7. None : keyword -> None
